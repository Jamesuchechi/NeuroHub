// Fibonacci with memoization
const memo = new Map();

function fib(n) {
  if (n <= 1) return n;
  if (memo.has(n)) return memo.get(n);
  const result = fib(n - 1) + fib(n - 2);
  memo.set(n, result);
  return result;
}

for (let i = 0; i <= 10; i++) {
  console.log(`fib(${i})`, fib(i));
}

console.log('done!');
